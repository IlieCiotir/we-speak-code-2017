Hide and show element functions 
<html>
    <head>
        <script>
            function showElement(id) {
                var element = document.getElementById(id);
                element.style.display = "block";
            }

            function hideElement(id) {
                var element = document.getElementById(id);
                element.style.display = "none";
            }
        </script>
    </head>
    <body>
        <div onclick="showElement('example-1-demo')">
            Show
        </div>
        <div onclick="hideElement('example-1-demo')">
            Hide
        </div>
        <div id="example-1-demo">
            Now you see me!
        </div>
    </body>
</html>

            margin-right: auto;
